# (a) Predict the long-term behavior
def predict_behavior(eigenvalues):
    predictions = []
    for lam in eigenvalues:
        if abs(lam) < 1:
            predictions.append("converge to zero")
        elif abs(lam) == 1:
            predictions.append("approach a cycle")
        else:
            predictions.append("become unbounded")
    return predictions

def plot_eigenspaces_and_points(eigenvectors, orbits):
    p = Graphics()
    
    # Plot eigenspaces
    for eigvec in eigenvectors:
        eigvec = vector(eigvec)
        eigvec_x, eigvec_y = eigvec[0], eigvec[1]
        p += parametric_plot((lambda t: t*eigvec_x, lambda t: t*eigvec_y), 
                             (-10, 10), linestyle="dashed")
    
    # Plot orbits
    for orbit in orbits:
        for i in range(len(orbit)-1):
            start, end = orbit[i],orbit[i+1]
            p += parametric_plot((lambda t:(1-t)*start[0]+t*end[0], 
                                  lambda t:(1 - t)*start[1]+t*end[1]), 
                                 (0, 1))
        # Plot points for the orbit
        for vec in orbit:
            p += point(vec, size=20)
    
    return p

# Orbit
def orbit(matrix, initial_vector, steps=50):
    orbit_points = [initial_vector]
    for _ in range(steps-1):
        new_point = matrix * orbit_points[-1]
        orbit_points.append(new_point)
    return orbit_points
