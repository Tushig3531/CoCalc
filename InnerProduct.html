    inner_products = [integrate(f*p, (x, -1, 1)) for p in basis]

def basis_0(x):
    return 1
def basis_1(x):
    return x
def basis_2(x):
    return (3*x**2-1)/2
def basis_3(x):
    return (5*x**3-3*x)/2
def basis_4(x):
    return (35*x**4-30*x**2+3)/8

basis_funcs = [basis_0, basis_1, basis_2, basis_3, basis_4]

def f1(x):
    return abs(x)
def f2(x):
    return np.cos(np.pi*x)
def f3(x):
    return np.sin(np.pi*x)

norms = [2, 2/3, 0.4, 0.2857, 0.2222]

# Projection
def compute_projection_function(f, basis_funcs, norms):
    inner_products = [] 
    for b in basis_funcs:
        result = quad(lambda x: f(x)*b(x), -1, 1)[0]
        inner_products.append(result)

    coeffs = []
    for inner, norm in zip(inner_products, norms): 
        coeff = inner/norm  
        coeffs.append(coeff)
        
    def projection(x):
        total = 0
        for a, b in zip(coeffs, basis_funcs): 
            total += a*b(x)
        return total
    return projection 

# Distance
def compute_distance(f, proj, is_absolute=False):
    if is_absolute:
        diff1 = quad(lambda x: (f(x) - proj(x))**2, 0, 1)[0]
        diff2 = quad(lambda x: (-f(x) - proj(x))**2, -1, 0)[0]
        return np.sqrt(diff1 + diff2)
    else:
        diff = quad(lambda x: (f(x) - proj(x))**2, -1, 1)[0]
        return np.sqrt(diff)
    
proj_f1 = compute_projection_function(f1, basis_funcs, norms)

# Indirect approach using Fourier series:
# f1(x) = x = 2(sin(x)/1-sin(3x)/3+sin(5x)/5 - ...)
# Only sine terms appear because x is an odd function.
# Using orthonormality, <x,x> = Sum(c_n^2) where c_n are the sine coefficients.
#
# From the known expansion, c_(2k-1) = 2(-1)^{k-1}/(2k-1), and c_(even)=0.
# Thus <x,x> = Sum(2/(2k-1))2 = 4 Sum(1/(2k-1)^2).
# Relating this to Sum(1/n2) leads to sum(1/n^2) = ^2/6.
